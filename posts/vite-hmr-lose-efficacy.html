  return router;
}

type History = {
  push: Router['navigate'];
  go: Router['navigate'];

      };

    return target[p]
  },
  set(target, p: keyof History, newValue, receiver) {
    target[p] = newValue;
    return true
  },
});

  history.push = router.navigate;
  history.go = router.navigate;
  history.back = () => router.navigate(-1);

export default history;
